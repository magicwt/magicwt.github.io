struct Queue
{

int front;
int rear;
int num;
};

void InitQueue(Queue &q)
{
q.front = q.rear = 99;
q.num = 0;
}

int EnQueue(Queue &q,Node* n)
{
if (q.num == 100) return -1;
q.rear = (q.rear+1)%100;
q.list[q.rear] = n;
q.num = q.num+1;
return 1;
}

int DeQueue(Queue &q,Node* &n)
{
if (q.num == 0) return -1;
q.front = (q.front+1)%100;
n = q.list[q.front];
q.num = q.num-1;
return 1;
}
